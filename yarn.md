# yarn
```
// 安装
$ brew install yarn
$ brew install yarn --ignore-dependencies

// 升级
$ brew upgrade yarn

// ???
$ npm i yarn -g

// 添加一个依赖
$ yarn add express

// 删除一个依赖
$ yarn remove express

// 安装一个开发依赖
$ yarn add gulp --dev

$ yarn why gulp
```

```
$ yarn -v
$ yarn -help

$ yarn init
$ yarn init --yes

$ yarn config set init-license ISC

// 安装特定版本的lodash
$ yarn add lodash@4.17.3


$ yarn outdated

// 全局安装
$ yarn global add nodemon
$ yarn global remote nodemon

$ yarn list
$ yarn list --depth=0
```

```
$ yarn run dev
```
